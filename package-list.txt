# C60 slim rootfs package list.
# One package per line. Blank lines and # comments ignored.
# Installed inside the chroot with `apt-get install --no-install-recommends`.

systemd
systemd-sysv
systemd-resolved
systemd-timesyncd
udev
dbus
libnss-systemd

# networking
iproute2
iputils-ping
isc-dhcp-client
openssh-server
wpasupplicant
iw
wireless-regdb
bluez

# wayland kiosk: compositor + browser
cage
cog
libwpebackend-fdo-1.0-1
libwpewebkit-1.1-0
libwpe-1.0-1
xwayland

# input / display / GPU
seatd
libseat1
libinput-bin
libinput10
libegl1
libgles2
mesa-utils
# Mesa DRI drivers (etnaviv_dri.so for the Vivante GC520 + kms_swrast fallback)
# and the Mesa EGL/GLX vendor libs are Recommends of libegl1/libglx0, so they
# are DROPPED by --no-install-recommends. List them explicitly or wlroots/cage
# GL init fails on boot and the kiosk never comes up.
libgl1-mesa-dri
libegl-mesa0
libglx-mesa0
libgbm1
# wayland client/server libs — hard-deps of cage/cog, pinned explicitly so the
# kiosk stack is self-contained and auditable.
libwayland-client0
libwayland-server0
libwayland-cursor0
libwayland-egl1
# shared MIME database: without it WPE/GIO detects file:///*.html as text/plain
# and the panel shows raw HTML source instead of the rendered touch-test page.
shared-mime-info

# minimal utils
util-linux
psmisc
procps
less
curl
ca-certificates
busybox-static

# locales / clock polish
locales
fake-hwclock

# u-boot environment tools (fw_printenv / fw_setenv) for reading and
# editing the u-boot env from running Linux via /etc/fw_env.config
# (written by chroot-setup.sh).
u-boot-tools
libubootenv-tool

# C60-specific extras
alsa-utils
ffmpeg
python3
